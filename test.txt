5)	замыкания – понимание:
функция  внутри  функции
	замыкание – пример:
		const  func  =  function()  {
			a  =  ‘’Hello  ‘’
			const  func2  =  function(x)  {
				console.log(  a  +  x  )
			}
			return  func2
		}
		let  func3  =  func()
		func3(‘’World’’)
		//func3  выводит  в  консоль  Hello  World
